    /* An order cannot move to (or past) dispatch without a rider attached.
       This is what prevents the "out_for_delivery with NULL rider" bug. */
    if (['assigned_to_delivery', 'out_for_delivery', 'arriving_soon'].includes(dto.status)
        && !order.deliveryPartnerId) {
      throw new BadRequestException(
        'Assign a delivery partner before moving this order out for delivery.');

    /* Bug #39: once a rider is assigned, forward progress toward delivery is
       the rider's job. An admin may still cancel or drive the pre-assignment
       lifecycle, but may NOT force an assigned order to out_for_delivery /
       arriving_soon / delivered — the admin only gets notified on delivery. */
    const riderOwnedStatuses = ['out_for_delivery', 'arriving_soon', 'delivered'];
    if (isAdmin && order.deliveryPartnerId && riderOwnedStatuses.includes(dto.status)) {
      throw new BadRequestException(
        'This order is assigned to a delivery partner. The rider updates delivery ' +
        'status from here — the admin will be notified when it is delivered.');
